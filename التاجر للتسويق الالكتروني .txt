  <script>
    (function(){
      // تعديل هذا إذا أردت رفع الملفات إلى باك-إند خاص بك ثم تضمين رابط الملف في رسالة الواتساب.
      // مثال: const UPLOAD_ENDPOINT = 'https://example.com/api/upload';
      const UPLOAD_ENDPOINT = ''; // اتركه فارغاً لإرسال الرسالة عبر واتساب فقط (بدون رفع تلقائي)

      const PHONE_WHATSAPP = '9647811831007'; // رقم المستلم لواتساب (بدون +)
      const OLD_PRICE = 70000;
      const NEW_PRICE = 60000;
      const PRODUCT_NAME = 'جهاز بايدر رياضي — قدم ووايدي متعدد الاستخدام';

      const fileInput = document.getElementById('fileInput');
      const previewArea = document.getElementById('previewArea');
      const fileError = document.getElementById('fileError');
      const mainImage = document.getElementById('mainImage');
      const oldPriceDisplay = document.getElementById('oldPriceDisplay');
      const newPriceDisplay = document.getElementById('newPriceDisplay');
      const discountBadge = document.getElementById('discountBadge');

      // عرض الأسعار وحساب الخصم
      function formatIQD(n){
        return n.toLocaleString('ar-EG') + ' د.ع';
      }
      oldPriceDisplay.textContent = formatIQD(OLD_PRICE);
      newPriceDisplay.textContent = formatIQD(NEW_PRICE);
      const discount = OLD_PRICE > 0 ? Math.round((OLD_PRICE - NEW_PRICE) / OLD_PRICE * 100) : 0;
      discountBadge.textContent = discount > 0 ? 'خصم ' + discount + '%' : 'سعر خاص';

      // ملفات: قيود الحجم
      const MAX_IMAGE = 5 * 1024 * 1024; // 5 MB
      const MAX_VIDEO = 1 * 1024 * 1024 * 1024; // 1 GB

      let selectedFile = null;
      fileInput.addEventListener('change', function(e){
        fileError.textContent = '';
        previewArea.innerHTML = '';
        selectedFile = null;
        const f = e.target.files[0];
        if(!f) return;
        const type = f.type || '';
        // image/*
        if(type.startsWith('image/')){
          if(f.size > MAX_IMAGE){
            fileError.textContent = 'حجم الصورة أكبر من 5 ميغابايت. الرجاء اختيار صورة أصغر.';
            fileInput.value = '';
            return;
          }
          const img = document.createElement('img');
          img.src = URL.createObjectURL(f);
          img.alt = 'معاينة الصورة';
          img.style.maxWidth = '100%';
          img.onload = () => URL.revokeObjectURL(img.src);
          previewArea.appendChild(img);
          mainImage.src = img.src;
          selectedFile = f;
        } else if(type.startsWith('video/')){
          if(f.size > MAX_VIDEO){
            fileError.textContent = 'حجم الفيديو أكبر من 1 جيجابايت. الرجاء اختيار ملف أصغر.';
            fileInput.value = '';
            return;
          }
          const vid = document.createElement('video');
          vid.controls = true;
          vid.src = URL.createObjectURL(f);
          vid.style.maxWidth = '100%';
          vid.onloadeddata = () => URL.revokeObjectURL(vid.src);
          previewArea.appendChild(vid);
          // ضع صورة افتراضية مع أيقونة الفيديو
          mainImage.src = 'https://via.placeholder.com/760x500?text=ملف+فيديو+مرفق';
          selectedFile = f;
        } else {
          fileError.textContent = 'نوع الملف غير مدعوم. اختر صورة أو فيديو (MP4 مفضّل).';
          fileInput.value = '';
        }
      });

      // حفظ/استعادة من localStorage
      const form = document.getElementById('orderForm');
      const storageKey = 'hb_order_form_v1';
      function saveForm(){
        const data = {
          name: form.name.value || '',
          phone: form.phone.value || '',
          governorate: form.governorate.value || '',
          nearest: form.nearest.value || '',
          address: form.address.value || ''
        };
        localStorage.setItem(storageKey, JSON.stringify(data));
      }
      function loadForm(){
        try{
          const raw = localStorage.getItem(storageKey);
          if(!raw) return;
          const d = JSON.parse(raw);
          form.name.value = d.name || '';
          form.phone.value = d.phone || '';
          form.governorate.value = d.governorate || '';
          form.nearest.value = d.nearest || '';
          form.address.value = d.address || '';
        }catch(e){}
      }
      form.addEventListener('input', saveForm);
      loadForm();

      // بناء رسالة الحجز
      function buildMessage(fileUrl){
        const name = form.name.value.trim() || '-';
        const phone = form.phone.value.trim() || '-';
        const governorate = form.governorate.value || '-';
        const nearest = form.nearest.value.trim() || '-';
        const address = form.address.value.trim() || '-';
        const lines = [
          'طلب حجز منتج',
          'المنتج: ' + PRODUCT_NAME,
          'السعر الحالي: ' + formatIQD(NEW_PRICE),
          'السعر السابق: ' + formatIQD(OLD_PRICE),
          'نسبة الخصم: ' + discount + '%',
          '',
          'معلومات المشتري:',
          'الاسم: ' + name,
          'الهاتف: ' + phone,
          'المحافظة: ' + governorate,
          'أقرب نقطة دالة: ' + nearest,
          'العنوان: ' + address,
          '',
          'التوصيل: مجاني لكل العراق',
        ];
        if(fileUrl){
          lines.push('');
          lines.push('رابط المرفق: ' + fileUrl);
        } else if(selectedFile){
          lines.push('');
          lines.push('ملاحظة: يوجد مرفق (صورة/فيديو) — رجاء إرفاقه يدوياً في محادثة الواتساب أو رفعه عبر رابط لكم.');
        }
        lines.push('');
        lines.push('الرجاء تأكيد الحجز والوقت المتاح للتوصيل.');
        return lines.join('\n');
      }

      // إرسال إلى واتساب (أو رفع أولاً إذا كان لدينا ENDPOINT)
      async function sendWhatsApp(){
        fileError.textContent = '';
        // تحقق من الحقول المطلوبة
        if(!form.name.value.trim()){
          alert('الرجاء إدخال الاسم الكامل.');
          form.name.focus(); return;
        }
        if(!form.phone.value.trim()){
          alert('الرجاء إدخال رقم الهاتف (واتساب).');
          form.phone.focus(); return;
        }
        if(!form.governorate.value){
          alert('الرجاء اختيار المحافظة.');
          form.governorate.focus(); return;
        }
        if(!form.address.value.trim()){
          alert('الرجاء إدخال العنوان التفصيلي.');
          form.address.focus(); return;
        }

        // إذا مُعرّف ENDPOINT قم بالرفع أولاً
        if(UPLOAD_ENDPOINT && selectedFile){
          try{
            const fd = new FormData();
            fd.append('file', selectedFile);
            // يمكنك إضافة حقول إضافية إن رغبت
            const res = await fetch(UPLOAD_ENDPOINT, { method: 'POST', body: fd });
            if(!res.ok) throw new Error('فشل رفع الملف. الرمز: ' + res.status);
            const json = await res.json();
            const fileUrl = json.url || json.file || '';
            const message = buildMessage(fileUrl);
            openWhatsAppWindow(message);
            return;
          } catch(err){
            fileError.textContent = 'خطأ في رفع الملف: ' + err.message;
            // استمر لإرسال الرسالة مع تعليمات إرفاق المرفق يدوياً
          }
        }

        // لا رفع => سنتيح الفتح مع رسالة جاهزة، ونطلب من المستخدم إرفاق الملف يدوياً إذا وُجد
        const message = buildMessage('');
        openWhatsAppWindow(message);
      }

      function openWhatsAppWindow(message){
        const encoded = encodeURIComponent(message);
        const url = 'https://wa.me/' + PHONE_WHATSAPP + '?text=' + encoded;
        // حاول فتح نافذة جديدة (ويب أو تطبيق)
        window.open(url, '_blank');
      }

      // نسخ نص الحجز
      document.getElementById('copyMessage').addEventListener('click', function(){
        const message = buildMessage('');
        navigator.clipboard.writeText(message).then(()=>{
          alert('نُسخ نص الحجز. يمكنك لصقه في واتساب وإرفاق الملف إن وُجد.');
        }).catch(()=>{
          alert('تعذّر نسخ النص تلقائياً. انسخ النص يدوياً:\n\n' + message);
        });
      });

      document.getElementById('sendWhatsapp').addEventListener('click', function(){
        sendWhatsApp();
      });

      document.getElementById('clearForm').addEventListener('click', function(){
        if(confirm('هل تود مسح بيانات النموذج؟')){
          form.reset();
          fileInput.value = '';
          previewArea.innerHTML = '';
          mainImage.src = 'https://via.placeholder.com/760x500?text=جهاز+بايدر+رياضي';
          selectedFile = null;
          localStorage.removeItem(storageKey);
        }
      });

      // تحسين تجربة: إذا ضغط المستخدم رقم الهاتف بزر Enter يتم فتح واتساب
      document.getElementById('phone').addEventListener('keydown', function(e){
        if(e.key === 'Enter'){ e.preventDefault(); sendWhatsApp(); }
      });

      // أضف بيانات schema.org الأساسية
      (function addSchema(){
        const productEl = document.querySelector('[itemscope][itemtype="https://schema.org/Product"]');
        if(!productEl) return;
        const offer = document.createElement('div');
        offer.innerHTML = '<meta itemprop="priceCurrency" content="IQD"><meta itemprop="price" content="' + NEW_PRICE + '"><link itemprop="availability" href="https://schema.org/InStock">';
        productEl.appendChild(offer);
      })();
    })();
  </script>
